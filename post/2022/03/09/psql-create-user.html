postgresql 创建用户并授权

Mar 09, 2022
postgresql
25 words in 1 min

# 创建用户

CREATE USER test WITH PASSWORD '*****';

# 创建数据库

create DATABASE dbTest;

# 将数据库的权限全部授予用户

GRANT ALL PRIVILEGES ON DATABASE dbTest TO test;
